    if (!title) return;

    const desc = resultBox.innerText.replace(title,'').trim();

    const payload = {
      lesson_id     : LESSON_ID,
      user_id       : USER_ID,
      project_title : title,
      project_desc  : desc,
      attempts_left : MAX_ATTEMPTS - attempts,
      timestamp     : new Date().toISOString()
    };

    try {
      if (USE_GET) {
        const qs = new URLSearchParams(payload).toString();
        await fetch(`${SAVE_ENDPOINT}?${qs}`, { mode:'cors' });
      } else {
        await fetch(SAVE_ENDPOINT, {
          method  : 'POST',
          mode    : 'cors',
          headers : { 'Content-Type':'application/json' },
          body    : JSON.stringify(payload)
        });
      }

        localStorage.removeItem(SAVED_KEY);     // сбросили флаг

        localStorage.setItem(CHOICE_KEY, html);
      });
    });
  }

    return `<strong class="proj-title">${p.title}</strong><br>` +

  }
